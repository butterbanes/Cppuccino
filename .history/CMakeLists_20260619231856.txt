cmake_minimum_required(VERSION 3.28)
project(Cppuccino VERSION 0.1
                  DESCRIPTION "Java-primitive Wrappers implemented in C++"
                  LANGUAGES CXX
)

include(GNUInstallDirs)

add_library(Cppuccino)

add_library(Cppuccino::Cppuccino ALIAS Cppuccino)

target_sources(Cppuccino PUBLIC
    FILE_SET HEADERS
        BASE_DIRS include
        FILES
            include/Cppuccino/Bool.h
            include/Cppuccino/Byte.h
            include/Cppuccino/Character.h
            include/Cppuccino/Double.h
            include/Cppuccino/Float.h
            include/Cppuccino/Integer.h
            include/Cppuccino/Long.h
            include/Cppuccino/Short.h
)

target_sources(Cppuccino PRIVATE
    src/Bool.cpp
    src/Byte.cpp
    src/Character.cpp
    src/Double.cpp
    src/Float.cpp
    src/Integer.cpp
    src/Long.cpp
    src/Short.cpp
)

target_include_directories(Cppuccino PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>
)

target_compile_features(Cppuccino PUBLIC cxx_std_20)

install(TARGETS Cppuccino
    EXPORT CppuccinoTargets
    FILE_SET HEADERS
)

install(EXPORT CppuccinoTargets
    FILE CppuccinoTargets.cmake
    NAMESPACE Cppuccino::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/Cppuccino
)
